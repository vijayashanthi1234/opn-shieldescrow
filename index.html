async function connectWallet() { if (typeof window.ethereum !== 'undefined') { try { // Request accounts explicitly const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' }); // Check if MetaMask returned an empty list if (!accounts || accounts.length === 0) { alert("No active account detected. Please open MetaMask and select an account."); return; } // Force connection to the first active account in the array const activeAccount = accounts[0]; provider = new ethers.providers.Web3Provider(window.ethereum); signer = provider.getSigner(); document.getElementById('walletAddress').innerText = activeAccount; const statusText = document.getElementById('walletStatus'); statusText.innerText = "Connected"; statusText.style.color = "var(--neon-green)"; document.getElementById('indStatus').className = "indicator ind-active"; console.log("Successfully locked to account:", activeAccount); } catch (error) { console.error("Connection Error:", error); alert("Connection Refused. Internal Code: " + (error.message || error.code || "Unknown")); } } else { alert("MetaMask extension provider not found."); } }
